# 쿼리 유형별 패턴 정의
# ===============================
QUERY_PATTERNS = {
    "CREATE_VIEW": re.compile(
        r"""
        \bCREATE\s+
        (?:OR\s+REPLACE\s+)?          
        (?:TEMPORARY\s+|TEMP\s+)?     
        VIEW\s+
        (?:IF\s+NOT\s+EXISTS\s+)?     
        ((?:`[^`]+`|"[^"]+"|\w+)(?:\.(?:`[^`]+`|"[^"]+"|\w+))?)
        """,
        re.IGNORECASE | re.VERBOSE
    ),
    "CTAS": re.compile(
        r"\bCREATE\s+TABLE\s+((?:`[^`]+`|\"[^\"]+\"|\w+))\s+AS\s+SELECT",
        re.IGNORECASE
    ),
    "INSERT": re.compile(
        r"\bINSERT\s+(?:OVERWRITE\s+)?INTO\s+((?:`[^`]+`|\"[^\"]+\"|\w+)(?:\.(?:`[^`]+`|\"[^\"]+\"|\w+))?)",
        re.IGNORECASE
    ),
    "UPDATE": re.compile(
        r"\bUPDATE\s+((?:`[^`]+`|\"[^\"]+\"|\w+))",
        re.IGNORECASE
    ),
    "DELETE": re.compile(
        r"\bDELETE\s+FROM\s+((?:`[^`]+`|\"[^\"]+\"|\w+))",
        re.IGNORECASE
    ),
    "MERGE": re.compile(
        r"\bMERGE\s+INTO\s+((?:`[^`]+`|\"[^\"]+\"|\w+))", re.IGNORECASE
    ),
    "WITH_CTE": re.compile(
        r"\bWITH\s+(\w+)\s+AS\s*\(", re.IGNORECASE
    ),
    "JOIN": re.compile(
        r"\b(INNER|LEFT|RIGHT|FULL|SELF)\s+JOIN\s+((?:`[^`]+`|\"[^\"]+\"|\w+))", re.IGNORECASE
    ),
    "FROM": re.compile(
        r"\bFROM\s+((?:\$\{\w+\}\.)?(?:`[^`]+`|\"[^\"]+\"|\w+))", re.IGNORECASE
    )
}

# ===============================
# 파일 확장자
# ===============================
FILE_EXTENSIONS = [".sh", ".hql", ".sql", ".uld", ".ld"]

# ===============================
# 스키마 변수 패턴
# ===============================
SCHEMA_PATTERN = re.compile(r"\$\{(\w+)\}")

# ===============================
# db_schema.env 읽기 (실행 디렉토리 기준)
# ===============================
def load_schema_env():
    schema_map = {}
    env_file_path = os.path.join(os.getcwd(), "db_schema.env")
    if os.path.exists(env_file_path):
        with open(env_file_path, "r") as f:
            for line in f:
                m = re.match(r'(\w+)\s*=\s*"([^"]+)"', line)
                if m:
                    schema_map[m.group(1)] = m.group(2)
    else:
        print(f"[WARN] db_schema.env 파일이 실행 디렉토리에 없습니다: {env_file_path}")
    return schema_map

# ===============================
# 소스 디렉토리 재귀 스캔
# ===============================
def scan_source_files(source_dir):
    files = []
    for root, dirs, filenames in os.walk(source_dir):
        for file in filenames:
            if os.path.splitext(file)[1].lower() in FILE_EXTENSIONS:
                files.append(os.path.join(root, file))
    return files

# ===============================
# 테이블/뷰 정보 추출
# ===============================
def extract_table_info(sql_text, schema_map):
    results = []
    for qtype, pattern in QUERY_PATTERNS.items():
        for match in pattern.finditer(sql_text):
            table_full = match.group(1)
            schema_real, table_name_only = "", table_full

            # ${스키마} 치환
            schema_var = SCHEMA_PATTERN.search(table_full)
            if schema_var:
                var_name = schema_var.group(1)
                schema_real = schema_map.get(var_name, var_name)
                table_name_only = table_full.split("}",1)[1].lstrip(".")
            elif "." in table_full:
                schema_real, table_name_only = table_full.split(".",1)

            results.append({
                "sql_type": qtype,
                "source_schema": schema_real,
                "source_table": table_name_only
            })
    return results

# ===============================
# CSV 생성
# ===============================
def write_csv(output_path, rows):
    os.makedirs(os.path.dirname(output_path), exist_ok=True)
    with open(output_path, "w", newline="", encoding="utf-8-sig") as f:
        writer = csv.writer(f)
        headers = [
            "file_path", "file_name", "full_file_path", "read_write",
            "sql_type", "source_schema", "source_table", "target_schema", "target_table"
        ]
        writer.writerow(headers)
        for row in rows:
            writer.writerow(row)

# ===============================
# 메인 실행
# ===============================
def main():
    if len(sys.argv) != 3:
        print("Usage: python3 program.py <source_dir> <mysql_table_name>")
        sys.exit(1)

    source_dir = sys.argv[1]
    MYSQL_TABLE = sys.argv[2]  # 입력받은 MySQL 테이블명

    schema_map = load_schema_env()
    all_files = scan_source_files(source_dir)
    output_rows = []

    for file_path in all_files:
        file_name = os.path.basename(file_path)
        with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
            sql_text = f.read()

        table_infos = extract_table_info(sql_text, schema_map)

        for info in table_infos:
            # READ/WRITE 구분
            write_types = ["INSERT", "MERGE", "UPDATE", "DELETE", "CTAS", "CREATE_VIEW"]
            rw_type = "WRITE" if info["sql_type"] in write_types else "READ"

            # 타겟 스키마/테이블 = source와 동일 (리니지용)
            tgt_schema, tgt_table = info["source_schema"], info["source_table"]

            output_rows.append([
                file_path, file_name, file_path, rw_type,
                info["sql_type"], info["source_schema"], info["source_table"], tgt_schema, tgt_table
            ])

    # CSV 파일명 생성
    last_dir = os.path.basename(os.path.normpath(source_dir))
    timestamp = datetime.datetime.now().strftime("%Y%m%d%H%M%S")
    csv_file = os.path.join(os.getcwd(), "out", f"{os.path.splitext(os.path.basename(sys.argv[0]))[0]}_{last_dir}_{timestamp}.csv")
    write_csv(csv_file, output_rows)

    # 상위 10개 결과 출력
    print("\n===== LINEAGE SAMPLE (TOP 10) =====")
    if output_rows:
        headers = ["file_path", "file_name", "full_file_path", "read_write",
                   "sql_type", "source_schema", "source_table", "target_schema", "target_table"]
        print(" | ".join(headers))
        print("-" * 150)
        for row in output_rows[:10]:
            print(" | ".join(row))
    else:
        print("분석 결과 없음")

    print(f"\n총 분석 건수: {len(output_rows)}")
    print(f"CSV 파일 생성: {csv_file}")
    print(f"MySQL 적재 테이블명: {MYSQL_TABLE}")
