#!/usr/bin/env python3
# -*- coding: utf-8 -*-

#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import os
import re
import csv
import sys
import datetime
# import pymysql  # MySQL 연결용, 필요 시 주석 해제

# ===============================
# MySQL 설정 예시 (주석처리)
# ===============================
# MYSQL_CONFIG = {
#     "host": "localhost",
#     "port": 3306,
#     "user": "your_user",
#     "password": "your_password",
#     "database": "your_database",
#     "charset": "utf8mb4",
#     "autocommit": False,
# }
# MYSQL_TABLE = "sql_lineage"

# ===============================
# 쿼리 유형별 패턴 정의
# ===============================
QUERY_PATTERNS = {
    "CREATE_VIEW": re.compile(
        r"""\bCREATE\s+(?:OR\s+REPLACE\s+)?(?:TEMPORARY\s+|TEMP\s+)?VIEW\s+(?:IF\s+NOT\s+EXISTS\s+)?((?:`[^`]+`|"[^"]+"|\w+)(?:\.(?:`[^`]+`|"[^"]+"|\w+))?)""",
        re.IGNORECASE | re.VERBOSE
    ),
    "INSERT_INTO": re.compile(r"\bINSERT\s+(?:OVERWRITE\s+)?INTO\s+((?:`[^`]+`|\"[^\"]+\"|\w+)(?:\.(?:`[^`]+`|\"[^\"]+\"|\w+))?)", re.IGNORECASE),
    "MERGE_UPSERT": re.compile(r"\bMERGE\s+INTO\s+((?:`[^`]+`|\"[^\"]+\"|\w+))", re.IGNORECASE),
    "UPDATE": re.compile(r"\bUPDATE\s+((?:`[^`]+`|\"[^\"]+\"|\w+))", re.IGNORECASE),
    "DELETE": re.compile(r"\bDELETE\s+FROM\s+((?:`[^`]+`|\"[^\"]+\"|\w+))", re.IGNORECASE),
    "CTAS": re.compile(r"\bCREATE\s+TABLE\s+((?:`[^`]+`|\"[^\"]+\"|\w+))\s+AS\s+SELECT", re.IGNORECASE),
    "WITH_CTE": re.compile(r"\bWITH\s+(\w+)\s+AS\s*\(", re.IGNORECASE),
    "JOIN": re.compile(r"\b(INNER|LEFT|RIGHT|FULL|SELF)\s+JOIN\s+((?:`[^`]+`|\"[^\"]+\"|\w+))", re.IGNORECASE),
}

# ===============================
# 확장자 필터
# ===============================
FILE_EXTENSIONS = [".sh", ".hql", ".sql", ".uld", ".ld"]

# ===============================
# 스키마 변수 추출 패턴
# ===============================
SCHEMA_PATTERN = re.compile(r"\$\{(\w+)\}")

# ===============================
# db_schema.env 읽기
# ===============================
def load_schema_env(env_file_path):
    schema_map = {}
    if os.path.exists(env_file_path):
        with open(env_file_path, "r") as f:
            for line in f:
                m = re.match(r'(\w+)\s*=\s*"([^"]+)"', line)
                if m:
                    schema_map[m.group(1)] = m.group(2)
    return schema_map

# ===============================
# 소스 디렉토리 내 파일 스캔
# ===============================
def scan_source_files(source_dir):
    file_list = []
    for root, dirs, files in os.walk(source_dir):
        for file in files:
            if os.path.splitext(file)[1].lower() in FILE_EXTENSIONS:
                file_list.append(os.path.join(root, file))
    return file_list

# ===============================
# 테이블 정보 추출
# ===============================
def extract_table_info(sql_text, schema_map):
    results = []
    for qtype, pattern in QUERY_PATTERNS.items():
        for match in pattern.finditer(sql_text):
            full_table_name = match.group(1)
            # 스키마 변수 치환
            schema_var_match = SCHEMA_PATTERN.match(full_table_name)
            if schema_var_match:
                var_name = schema_var_match.group(1)
                schema_value = schema_map.get(var_name, var_name)
                full_table_name = full_table_name.replace("${" + var_name + "}", schema_value)
            results.append({"sql_type": qtype, "table_name": full_table_name})
    return results

# ===============================
# CSV 생성
# ===============================
def write_csv(output_path, rows, include_comment=True):
    os.makedirs(os.path.dirname(output_path), exist_ok=True)
    with open(output_path, "w", newline="", encoding="utf-8-sig") as f:
        writer = csv.writer(f)
        headers = [
            "file_path", "file_name", "full_file_path", "read_write",
            "sql_type", "source_schema", "source_table", "target_schema", "target_table"
        ]
        if include_comment:
            headers.append("comment")
        writer.writerow(headers)
        for row in rows:
            if not include_comment:
                row = row[:-1]  # comment 컬럼 제거
            writer.writerow(row)

# ===============================
# 메인 실행
# ===============================
def main():
    if len(sys.argv) != 3:
        print("Usage: python3 sql_lineage_scanner.py source_dir target_table")
        sys.exit(1)

    source_dir = sys.argv[1]
    target_table = sys.argv[2]
    env_file = os.path.join(os.path.dirname(source_dir), "db_schema.env")
    schema_map = load_schema_env(env_file)

    all_files = scan_source_files(source_dir)
    output_rows = []

    for file_path in all_files:
        file_name = os.path.basename(file_path)
        with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
            sql_text = f.read()

        table_infos = extract_table_info(sql_text, schema_map)

        for info in table_infos:
            # READ/WRITE 구분
            write_types = ["INSERT_INTO", "MERGE_UPSERT", "UPDATE", "DELETE", "CTAS", "CREATE_VIEW"]
            rw_type = "WRITE" if info["sql_type"] in write_types else "READ"

            # 소스스키마 / 소스테이블
            if "." in info["table_name"]:
                src_schema, src_table = info["table_name"].split(".", 1)
            else:
                src_schema, src_table = "", info["table_name"]

            # 타겟스키마/테이블: 현재는 전달받은 테이블명으로 표시
            tgt_schema, tgt_table = "", target_table

            output_rows.append([
                file_path,
                file_name,
                file_path,
                rw_type,
                info["sql_type"],
                src_schema,
                src_table,
                tgt_schema,
                tgt_table,
                ""  # comment 비워두기
            ])

    # CSV 파일명 생성
    last_dir = os.path.basename(os.path.normpath(source_dir))
    timestamp = datetime.datetime.now().strftime("%Y%m%d%H%M%S")
    csv_file = os.path.join(os.getcwd(), "out", f"{os.path.splitext(os.path.basename(sys.argv[0]))[0]}_{last_dir}_{timestamp}.csv")
    write_csv(csv_file, output_rows, include_comment=True)
    print(f"CSV 생성 완료: {csv_file}")

if __name__ == "__main__":
    main()
